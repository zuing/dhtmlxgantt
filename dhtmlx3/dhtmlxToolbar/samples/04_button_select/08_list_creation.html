<script>
var sel;
var sel2;
var toolbar;
function doOnLoad() {
    sel = document.getElementById("sel");
    sel2 = document.getElementById("sel2");
    toolbar = new dhtmlXToolbarObject("toolbarObj");
    toolbar.setIconsPath("../common/imgs/");
    toolbar.loadXML("../common/dhxtoolbar_buttonselect.xml?etc=" + new Date().getTime(), updateList);
}
function getId(selObj) {
    var id = selObj.options[selObj.selectedIndex].value;
    return id;
}
var img = "text_document.gif";
var lastImg = document.getElementById("text_document");
function changeImage(imgObj) {
    img = imgObj.id;
    imgObj.style.border = "#707070 1px solid";
    lastImg.style.border = "#E0E0E0 1px solid";
    lastImg = imgObj;
}
function remove() {
    if (sel2.options.length == 0) {
        return;
    }
    var id = getId(sel2).split(";");
    toolbar.removeListOption(id[0], id[1]);
    updateList();
}
function add() {
    var itemId = getId(sel);
    var pos = Number(document.getElementById("txt1").value);
    var text = document.getElementById("txt2").value;
    id = new Date().getTime();
    toolbar.addListOption(itemId, id, pos, "button", text, lastImg.id + ".gif");
    updateList();
}
function updateList() {
    sel.options.length = 0;
    sel2.options.length = 0;
    toolbar.forEachItem(function(itemId) {
        if (toolbar.getType(itemId) == "buttonSelect") {
            sel.options.add(new Option(itemId, itemId));
            toolbar.forEachListOption(itemId, function(listId) {
                sel2.options.add(new Option(itemId + " - " + listId, itemId + ";" + listId));
            });
        }
    });
}
</script>
