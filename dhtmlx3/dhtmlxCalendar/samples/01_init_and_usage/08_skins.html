var myCalendar;
function doOnLoad() {
    myCalendar = new dhtmlXCalendarObject("calendarHere");
    myCalendar.setSkin("dhx_skyblue");
    myCalendar.setDate("2011-05-17");
    myCalendar.show();
    //myCalendar.setHolidays(["2011-05-01","2011-05-09","2011-04-29","2011-04-30"]);
    myCalendar.setSensitive(["2011-05-27", "2011-05-21", "2011-05-28", "2011-06-03", "2011-06-04", "2011-05-20", "2011-05-13", "2011-05-14", "2011-04-29", "2011-04-30"]);
    myCalendar.setHolidays(["2011-04-28", "2011-05-09", "2011-05-01", "2011-05-27", "2011-05-21", "2011-05-28", "2011-06-03", "2011-06-04"]);
    myCalendar._drawMonth(new Date(2011, 4, 1));
}
</script>
